## Install

```bash
npm install --save use-pagination-hook
```

## Usage

```tsx
import * as React from 'react'

import { useMyHook } from 'use-pagination-hook'

const Example = () => {
  const example = useMyHook()
  return (
    <div>
      {example}
    </div>
  )
}
```
